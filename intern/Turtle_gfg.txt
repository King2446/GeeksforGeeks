import turtle as t
def drawfib(n, len_ang):
	t.forward(2 * len_ang)
	if n == 0:
		pass
	elif n == 1:
		pass
	else:
		t.left(len_ang)
		drawfib(n - 1, len_ang)
		t.right(2 * len_ang)
		drawfib(n - 2, len_ang)
		t.left(len_ang)
	t.backward(2 * len_ang)
# Six different starting points for six different trees.
start_points = [[-300, 250], [-150, 250],[-300, 110], [-80, 110],[-300, -150], [50, -150]]
# For each starting point, draw a tree with n varying
# between 1 and 6 and len_ang set to 30.
n = 0
for start_point in start_points:
	x, y = start_point
	n = n + 1
	t.penup()
	t.setpos(x, y)
	t.pendown()
	drawfib(n, 30)

# Making Colorful Spiral Web With Python.
colors = []

# Input variables:
print ("Please enter the 'X' & 'Y' coordinates of the canvas.")
X = int (input ("\tPlease enter the 'X' coordinate of the canvas: "))
Y = int (input ("\tPlease enter the 'Y' coordinate of the canvas: "))
R = int (input ("Please enter the 'Range' value of the spiral: "))
S = int (input ("Please enter the speed limit of the turtle: "))
SN = int (input ("Please define the number of sides of the base poligon: "))
A = int (input ("Please define the degree of angle of the base poligon: "))
print ("Please enter 'Y' or 'y', if you want the spiral to speen in 'Clockwise' direction, else enter 'N' or 'n'.")
SPEEN = input ("\tPlease enter your choice: ")
Back_Ground = input ("Please enter the Back Ground color of the spiral: ")

for i in range (SN):
	Color = input ("Please enter the colors you want to see in the spiral: ")
	colors.append (Color)

if SPEEN == "Y" or SPEEN == "y":
	Speen = 'Clockwise'

elif SPEEN == "N" or SPEEN == "n":
	Speen = 'Anti-Clockwise'

else:
	print ("\n Please re-check and re-enter the command.")

import turtle
wn = turtle.Screen()
screen=turtle.Screen()
t = turtle.Turtle()

screen.setup(X,Y)
screen.bgcolor(Back_Ground)

t.speed (S)
t.ht()

Pal = len (colors)
Love = Pal - (Pal - SN)
Dil = R % Love

for x in range(R):
	t.pencolor(colors[x%SN])
	t.width(x/100 + 1)
	t.forward(x)
	if Speen == "Clockwise":
		t.right(A)

	elif Speen == "Anti-Clockwise":
		t.left(A)

t.pencolor(colors[Dil])
t.forward(R/2)

turtle.done()
